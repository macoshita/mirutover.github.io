require 'thor'

class Hoge < Thor
  desc "fuga YYYYMMDD", "fugafuga"
  def fuga(yyyymmdd)
    raise HogeError, "invalid date format" unless /^\d{8}$/ =~ yyyymmdd

    puts "ok!"

  rescue HogeError => ex
    puts ex
  end
  
  private
  class HogeError < StandardError
  end
end
